% Auto-show author emails and cover image on the PDF title page
\usepackage{graphicx}
\makeatletter
\renewcommand{\maketitle}{
  \begin{titlepage}
    \centering
    {\Huge\bfseries \@title \par}
    $if(subtitle)$
      \vspace{0.5cm}{\Large \@subtitle \par}
    $endif$

    \vspace{1cm}
    {\Large \@author \par}

    % Emails pulled from YAML (supports multiple authors)
    $if(author)$
      \vspace{0.3cm}
      {\large \texttt{$for(author)$$if(author.email)$$author.email$$endif$$sep$ \quad\textbullet\quad $endfor$}\par}
    $endif$

    \vfill

    % Cover image if provided in YAML
    $if(cover-image)$
      {\includegraphics[width=0.4\textwidth]{$cover-image$}\par}
      \vfill
    $endif$

    {\large \@date\par}
  \end{titlepage}
}
\makeatother
